\newcommand{\EducationEntry}[4]{
		\noindent \textbf{#1} \hfill 					% Study

			\parbox{10em}{%

		\noindent\hangindent=2em\hangafter=0 #3 \par					% School

		\normalsize \par}

		\noindent \small #4 	% Description
		\normalsize \par}
